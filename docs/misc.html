def per (t, p = 0.5):
  p = int(p * len(t) + 0.5)
  return t[max(1, min(len(t), p))]

def push(t, x):
  t[1 + len(t)] = x
  return x

def rnd(x, places = 2):
  return round(x, places)


def o(t):
  if (type(t) !=  dict and type(t).__module__ == "__builtin__"):
    return str(t)
  if(type(t).__module__ != "__builtin__" and type(t) != dict):
    newDictionary = (vars(t))
    newDictionary.pop('_has')
    return newDictionary

  print(t)
  def show(k,v):
    first = k[0]
    if(str(first)!="_"):
      if(type(v) == dict):
        v = o(v)
        if(len(t) == 0):
          return ":"+str(k)+str(v)
        else:
          return str(v)
  u={}
  for k,v in t.items():
    u_len = len(u)
    u[k] = show(k,v)
  if len(t)==0:
    u = sorted(u)
  output = ""
  for key in u:
    output = output + ":" + key + " " + str(u[key]) + " "
  return "{" + output + "}"
